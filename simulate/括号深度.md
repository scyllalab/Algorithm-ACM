```C++
#include<iostream>
using namespace std;

int main(){

    string input="[{()}()]";
    char strs[100000]={0};
    // cin>>input;
    cout<<input<<endl;
    int max=0, inValid=0,length=0;
    for(int i=0; i<input.length();i++){
        if(input[i]=='('||input[i]=='['||input[i]=='{'){
            strs[length++]=input[i];
            continue;
        }
        if(input[i]==')'&&length>=1){
            if(strs[length-1]=='('){
                if(max < length)
                    max = length;
                length--;
                continue;
            }else{
                inValid=1;
                break;
            }
        }
        if(input[i]==']'&&length>=1){
            if(strs[length-1]=='['){
                if(max < length)
                    max = length;
                length--;
                continue;
            }else{
                inValid=1;
                break;
            }
            
        }

        if(input[i]=='}'&&length>=1){
            if(strs[length-1]=='{'){
                if(max < length)
                    max = length;
                length--;
                continue;
            }else{
                inValid=1;
                break;
            }
            
        }
    }
    if(!inValid)
        cout<<max<<endl;
    else
        cout<<"0"<<endl;

    return 0;
}
```
